<script>

function calculateScores() {

    var days = parseFloat(document.getElementById('days').value);

    var updatesPerDay = parseFloat(document.getElementById('updatesPerDay').value);

    var ratio = parseFloat(document.getElementById('ratio').value);

    var notDelivered = parseFloat(document.getElementById('notDelivered').value);

    var oosFail = parseFloat(document.getElementById('oosFail').value);


    var scoreA = days <= 2 ? 10 : days <= 4 ? 7.5 : days <= 6 ? 5 : days <= 8 ? 2.5 : 0;

    var scoreB = updatesPerDay >= 3 ? 10 : updatesPerDay >= 2 ? 7.5 : updatesPerDay >= 1 ? 5 : updatesPerDay >= 0.5 ? 2.5 : 0;

    var scoreC = ratio >= 5.0 ? 10 : ratio >= 4.0 ? 7.5 : ratio >= 2.0 ? 5 : ratio >= 1.0 ? 2.5 : 0;

    var scoreD = notDelivered <= 1 ? 10 : notDelivered <= 2 ? 7.5 : notDelivered <= 5 ? 5 : notDelivered <= 10 ? 2.5 : 0;

    var scoreE = oosFail <= 1 ? 10 : oosFail <= 2 ? 7.5 : oosFail <= 5 ? 5 : oosFail <= 10 ? 2.5 : 0;


    var totalScore = ((scoreA + scoreB + scoreC + scoreD + scoreE) / 5) * 10;


    document.getElementById('result').innerText = "Total SFTP Score: " + totalScore.toFixed(2);

}

</script>
